import java.util.*;
class Solution {
    public Object solution(String s) {
        char[] charS = s.toCharArray();
        Arrays.sort(charS);
        String str = new String(charS);
        StringBuffer answer = new StringBuffer(str);
        return answer.reverse().toString();
    }
}
